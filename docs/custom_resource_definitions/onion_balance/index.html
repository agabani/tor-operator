Onion Balance

Examples

Basic

The Tor Operator will create an Onion Balance using an auto generated Onion Key load load balancing a list of Onion Services.

#onionkey.yaml
apiVersion: tor.agabani.co.uk/v1
kind: OnionKey
metadata:
  name: onion-balance-{{ include "example.fullname" . }}
  labels:
    {{- include "example.labels" . | nindent 4 }}
spec:
  auto_generate: true
  secret:
    name: onion-balance-{{ include "example.fullname" . }}


#onionbalance.yaml
apiVersion: tor.agabani.co.uk/v1
kind: OnionBalance
metadata:
  name: {{ include "example.fullname" . }}
  labels:
    {{- include "example.labels" . | nindent 4 }}
spec:
  config_map:
    name: onion-balance-{{ include "example.fullname" . }}
  deployment:
    name: onion-balance-{{ include "example.fullname" . }}
  onion_key:
    name: onion-balance-{{ include "example.fullname" . }}
  onion_services:
    - onion_key:
        hostname: 3evpjb3a5qxaubnzodlxlgn3ccp3achwwwtirgpznnl752oyg2oxekid.onion
